91  void Execute(const itk::Object * object, const itk::EventObject & event)
92  {
93  OptimizerPointer optimizer =
94  dynamic_cast< OptimizerPointer >( object );
95  if( ! itk::IterationEvent().CheckEvent( &event ) )
96  {
97  return;
98  }
99 
100  double value = optimizer->GetValue();
101 
102  if (logging) {
103  std::cout << optimizer->GetCurrentIteration() + 1 << " ";
104  std::cout << value << " ";
105  }
106  if ( optimizer->GetCurrentIteration() > 0 && value < bestValue ) {
107  bestValue = value;
108  OptimizerType::ParametersType param(optimizer->GetCurrentPosition());
109  bestParameters = param;
110  }
111  if (logging) {
112  std::cout << optimizer->GetCurrentPosition();
113  std::cout << std::endl;
114  }
115  }
