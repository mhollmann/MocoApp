97  void Execute(const itk::Object * object, const itk::EventObject & event)
98  {
99  OptimizerPointer optimizer =
100  dynamic_cast< OptimizerPointer >( object );
101  if( ! itk::IterationEvent().CheckEvent( &event ) )
102  {
103  return;
104  }
105 
106  double value = optimizer->GetValue();
107 
108  if (logging) {
109  std::cout << optimizer->GetCurrentIteration() + 1 << " ";
110  std::cout << value << " ";
111  }
112  if ( optimizer->GetCurrentIteration() > 0 && value < bestValue ) {
113  bestValue = value;
114  OptimizerType::ParametersType param(optimizer->GetCurrentPosition());
115  bestParameters = param;
116  }
117  if (logging) {
118  std::cout << optimizer->GetCurrentPosition();
119  std::cout << std::endl;
120  }
121  }
